const server = http.createServer((req,res) => {
    // code

express 中间件的使用

它是处理过程中的一个环节; 是一个函数

可以执行任何代码

可以修改请求和响应对象

可以终结请求-响应循环

可以调用堆栈中的下一个中间件

const express = require(‘express’)

const app = express()
